\section{Introdução}
% ------------------------------------------------------------

Um \textbf{Autômato Finito Determinístico (AFD)} é um modelo matemático de computação utilizado para reconhecer linguagens regulares. Ele processa uma sequência finita de símbolos de entrada, transitando entre estados de acordo com uma função de transição determinística, e ao final decide se a palavra lida pertence ou não à linguagem reconhecida. Formalmente, um AFD é caracterizado por cinco componentes:

\begin{enumerate}
    \item Um conjunto finito de \textbf{estados} denotado $Q$.
    \item Um conjunto finito de \textbf{símbolos de entrada} denotado por $\Sigma$.
    \item Uma \textbf{função de transição} $\delta$.
    \item Um \textbf{estado inicial} $q_0$, estado onde o autômato começa a computação.
    \item Um subconjunto de $Q$, denotado de $F$, de \textbf{estados de aceitação}.
\end{enumerate}

\section{Definição}

\begin{definition}[Autômato Finito Determinístico]
É definido por uma 5-tupla ordenada:

\[
M = (Q, \Sigma, \delta, q_0, F),
\]
onde:

\begin{itemize}
    \item $Q$ é um conjunto finito de estados;
    \item $\Sigma$ é o alfabeto de entrada;
    \item $\delta : Q \times \Sigma \to Q$ é a função de transição total;
    \item $q_0 \in Q$ é o estado inicial;
    \item $F \subseteq Q$ é o conjunto de estados de aceitação.
\end{itemize}

\end{definition}

\section{Implementação de um AFD}
% ------------------------------------------------------------

\begin{algorithm}[H]
  \caption{Simulação de um AFD com $\delta(\cdot, \cdot)$ completa}
  \label{alg:afdCompleto}
  \begin{algorithmic}[1]
    \Require Um AFD $M = (Q, \Sigma, \delta, q_0, F)$,
             onde $\delta : Q \times \Sigma \to Q$,
             e uma palavra $w \in \Sigma^*$
    \Ensure  \textbf{aceita} se $w \in L(M)$, \textbf{rejeita} caso contrário
    \State $q_{\text{atual}} \gets q_0$
    \For{cada símbolo $c$ em $w$}
      \State $q_{\text{atual}} \gets \delta(q_{\text{atual}}, c)$
    \EndFor
    \If{$q_{\text{atual}} \in F$}
      \State \Return \textbf{aceita}
    \Else
      \State \Return \textbf{rejeita}
    \EndIf
  \end{algorithmic}
\end{algorithm}

\section{O que é função de transição parcial?}

Observe que ao definir um AFD, usamos a \textbf{função de transição total}, o que implica que todo para todo par $(q, \sigma) \in Q \times \Sigma$ existe exatamente um estado destino $\delta(q, \sigma) \in Q$. Em alguns casos, algumas combinações $(q, \sigma) \in Q \times \Sigma$ simplesmente não têm transição definida, surge então a \textbf{função de transição parcial}:

\[
\delta : Q \times \Sigma \rightharpoonup Q
\]
Para esses casos, teríamos que corrigir o algoritmo ~\ref{alg:afdCompleto}, verificando se a transição existe:
\begin{algorithm}[H]
  \caption{Simulação de um AFD com $\delta(\cdot, \cdot)$ parcial}
  \label{alg:afdCompleto}
  \begin{algorithmic}[1]
    \Require Um AFD $M = (Q, \Sigma, \delta, q_0, F)$,
             onde $\delta : Q \times \Sigma \to Q$,
             e uma palavra $w \in \Sigma^*$
    \Ensure  \textbf{aceita} se $w \in L(M)$, \textbf{rejeita} caso contrário
    \State $q_{\text{atual}} \gets q_0$
    \For{cada símbolo $c$ em $w$}
      \If{$\delta(q_{\text{atual}}, c)$ está definido}
        \State $q_{\text{atual}} \gets \delta(q_{\text{atual}}, c)$
      \Else
        \State \Return \textbf{rejeita}
      \EndIf
    \EndFor
    \If{$q_{\text{atual}} \in F$}
      \State \Return \textbf{aceita}
    \Else
      \State \Return \textbf{rejeita}
    \EndIf
  \end{algorithmic}
\end{algorithm}
